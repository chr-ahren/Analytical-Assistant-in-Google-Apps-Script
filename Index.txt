function setStatus(m){document.getElementById('status').textContent='Estado: '+m;}

function _stamp(){
  var d=new Date();
  return [d.getFullYear(),String(d.getMonth()+1).padStart(2,'0'),String(d.getDate()).padStart(2,'0')].join('')
    +'_'+[String(d.getHours()).padStart(2,'0'),String(d.getMinutes()).padStart(2,'0')].join('');
}

  setStatus('Registrando referencia...');

      setStatus(r.message+' Función: '+r.functionName);

function loadDemoPython(){
  document.getElementById('functionName').value='limpieza';
  document.getElementById('pythonCode').value=[
    "import pandas as pd","import numpy as np","",
    "def limpieza(archivo):",
    "    data = pd.read_csv(archivo)",

    "    data['N1'] = np.round(np.clip(np.random.normal(3.5,0.5,len(data)),1,5),1)",
    "    data['N2'] = np.round(np.random.uniform(1,5,len(data)),1)",
    "    data['N3'] = np.round(np.random.uniform(4,5,len(data)),1)",

function readCsvFile(e){
  var f=e.target.files&&e.target.files[0]; if(!f) return;
  var r=new FileReader();
  r.onload=function(ev){document.getElementById('csvText').value=ev.target.result;setStatus('CSV cargado.');};
  r.onerror=function(){setStatus('❌ No se pudo leer el archivo.');};
  r.readAsText(f,'UTF-8');
}

function loadCsvPreview(){
  var csv=getCsvText(); if(!csv.trim()){setStatus('❌ Cargue o pegue el CSV.');return;}
  setStatus('Cargando datos...');
  google.script.run
    .withSuccessHandler(function(r){
      reportLog.cargaDatos={totalRows:r.totalRows,columns:r.columns,timestamp:nowLabel()};
      setStatus(r.message+' Registros: '+r.totalRows);
      renderBadges(['Filas: '+r.totalRows,'Columnas: '+r.columns.length]);
      renderTable(r.preview,'Vista previa de datos cargados');
    }).withFailureHandler(handleError).loadCsvData(csv);
}

function runCleaning(){
  var csv=getCsvText(); if(!csv.trim()){setStatus('❌ Cargue o pegue el CSV.');return;}
  setStatus('Ejecutando limpieza...');
  google.script.run
    .withSuccessHandler(function(r){
      lastCleanCsv=r.csvOutput; lastModelCsv=''; lastCleanPreview=r.preview||[];
      reportLog.limpieza={shape:r.shape,columns:r.columns,timestamp:nowLabel()};
      setStatus(r.message+' Filas: '+r.shape.rows+' | Cols: '+r.shape.columns);
      renderBadges(['Limpieza OK','Filas: '+r.shape.rows,'Cols: '+r.shape.columns]);
      renderTable(r.preview,'Datos limpios');
    }).withFailureHandler(handleError).runLimpieza(csv);
}

function runAnalysis(){
  var csv=getCsvText(); if(!csv.trim()){setStatus('❌ Cargue o pegue el CSV.');return;}
  setStatus('Generando análisis descriptivo...');
  google.script.run
    .withSuccessHandler(function(r){
      lastAnalysis=r;
      reportLog.analisis={totalRows:r.totalRows,stats:r.stats,groupsCount:r.groupsCount,timestamp:nowLabel()};
      setStatus(r.message);
      renderAnalysis(r);
    }).withFailureHandler(handleError).runDescriptiveAnalysis(csv);
}

function showAdaptation(){
  var csv=getCsvText(); if(!csv.trim()){setStatus('❌ Cargue o pegue el CSV.');return;}
  setStatus('Ejecutando K-Means...');
  google.script.run
    .withSuccessHandler(function(r){
      lastModelCsv=r.csvOutput; lastKMeansResult=r;
      reportLog.adaptacion={totalRows:r.totalRows,k:r.k,summary:r.summary,interpretation:r.interpretation,timestamp:nowLabel()};
      setStatus(r.message+' Registros: '+r.totalRows);
      renderKMeansResult(r);
    }).withFailureHandler(handleError).runKMeansAdaptation(csv);
}

/* ══════════════════════════════════════════
   📋 INFORME — genera PDF en memoria
══════════════════════════════════════════ */
async function buildReport(){
  var jsPDFCls = window.jspdf && window.jspdf.jsPDF;
  if(!jsPDFCls){setStatus('❌ jsPDF no disponible. Verifique conexión.');return;}
  var hasData=reportLog.cargaDatos||reportLog.limpieza||reportLog.analisis||reportLog.adaptacion;
  if(!hasData){setStatus('❌ Ejecute al menos un paso antes de generar el informe.');return;}
  document.getElementById('btnInforme').disabled=true;
  showBadge(false);
  setStatus('⏳ Redibujando gráficas para PDF y capturando… esto toma ~5 segundos.');
  try{
    var imgs=await captureAllCharts();
    var doc =await buildPDF(null, imgs);
    pdfBlob   =doc.output('blob');
    pdfFileName='Informe_Analitico_'+_stamp()+'.pdf';
    showBadge(true);
    setStatus('✅ Informe PDF listo. Haz clic en "Descargar CSV limpio" para obtener PDF + CSV.');
  }catch(err){
    setStatus('❌ Error al generar PDF: '+err.message);
    console.error(err);
  }
  document.getElementById('btnInforme').disabled=false;
}

/* ══════════════════════════════════════════
   ⬇ DESCARGAR CSV — descarga PDF + CSV
══════════════════════════════════════════ */
function downloadAll(){
  var downloaded=[];
  if(pdfBlob){
    _dl(URL.createObjectURL(pdfBlob), pdfFileName);
    downloaded.push('PDF');
  }
  var csvData='', csvName='';
  if(lastModelCsv){csvData=lastModelCsv; csvName='Resultado_KMeans_'+_stamp()+'.csv';}
  else if(lastCleanCsv){csvData=lastCleanCsv; csvName='Data_Limpia_'+_stamp()+'.csv';}
  if(csvData){
    var hdr=['# ============================================================',
      '# Asistente Analítico en Google Apps Script  v3.0.0',
      '# Exportado: '+nowLabel(),
      '# Desarrollado por: Ahren Kahi Churta',
      '# ============================================================',''].join('\n');
    _dl(URL.createObjectURL(new Blob([hdr+csvData],{type:'text/csv;charset=utf-8;'})), csvName);
    downloaded.push('CSV');
  }
  if(!downloaded.length){setStatus('❌ No hay informe ni datos. Ejecute los pasos primero.');return;}
  setStatus('✅ Descargado: '+downloaded.join(' + '));
}
function _dl(url,name){
  var a=document.createElement('a'); a.href=url; a.download=name;
  document.body.appendChild(a); a.click(); document.body.removeChild(a);
  setTimeout(function(){URL.revokeObjectURL(url);},4000);
}

/* ══════════════════════════════════════════
   CAPTURA DE GRÁFICAS — estrategia offscreen definitiva

   PROBLEMA RAÍZ resuelto:
   - Los elementos del DOM principal pueden tener offsetParent=null
     cuando el usuario cambió de vista (ej: estaba en K-Means y los
     charts de análisis ya no se ven). html2canvas falla en silencio.
   - El scatter de K-Means no llegaba a renderizar a tiempo.

   SOLUCIÓN:
   - Contenedor offscreen dedicado (#pdf-charts-offscreen) siempre
     en el DOM con position:absolute left:-9999px → offsetParent NUNCA null.
   - Se renderizan los 8 charts directamente en ese contenedor con
     opciones de fondo blanco.
   - Se espera 900ms (todos los charts se renderizan en paralelo).
   - html2canvas captura desde el contenedor offscreen, no desde la UI.
   - Los charts de la UI se restauran después sin bloquear el PDF.
══════════════════════════════════════════ */

/* Convierte un SVG DOM a dataURL PNG usando canvas nativo — sin html2canvas */
function svgToPng(svgEl, w, h){
  return new Promise(function(resolve){
    try{
      /* Serializar SVG a string */
      var serializer = new XMLSerializer();
      var svgStr = serializer.serializeToString(svgEl);

      /* Asegurar atributos de dimensión */
      if(svgStr.indexOf('xmlns=') === -1)
        svgStr = svgStr.replace('<svg','<svg xmlns="http://www.w3.org/2000/svg"');

      /* Crear Blob y Object URL */
      var blob = new Blob([svgStr],{type:'image/svg+xml;charset=utf-8'});
      var url  = URL.createObjectURL(blob);
      var img  = new Image();

      img.onload = function(){
        var cv  = document.createElement('canvas');
        cv.width  = w||820;
        cv.height = h||300;
        var ctx = cv.getContext('2d');
        ctx.fillStyle = '#ffffff';
        ctx.fillRect(0,0,cv.width,cv.height);
        ctx.drawImage(img,0,0,cv.width,cv.height);
        URL.revokeObjectURL(url);
        resolve(cv.toDataURL('image/png'));
      };
      img.onerror = function(){
        URL.revokeObjectURL(url);
        resolve(null);
      };
      img.src = url;
    }catch(e){
      console.warn('svgToPng error',e);
      resolve(null);
    }
  });
}

async function captureAllCharts(){
  var imgs = {};
  var pcIds = Object.keys(CHART_MAP);

  /* 1. Dibujar todos en offscreen a la vez */
  drawChartsOffscreen();

  /* 2. Esperar una sola vez a que Google Charts termine (900ms) */
  await new Promise(function(res){ setTimeout(res, 900); });

  /* 3. Convertir SVG → PNG para cada contenedor (rápido, sin bloqueo) */
  for(var i=0;i<pcIds.length;i++){
    var pcId = pcIds[i];
    var el   = document.getElementById(pcId);
    if(!el) continue;
    var svg  = el.querySelector('svg');
    if(!svg){ console.warn('sin SVG:',pcId); continue; }
    var w = el.offsetWidth  || 820;
    var h = el.offsetHeight || 300;
    var png = await svgToPng(svg, w, h);
    if(png) imgs[CHART_MAP[pcId]] = png;
  }

  /* 4. Limpiar offscreen */
  pcIds.forEach(function(id){
    var el=document.getElementById(id); if(el) el.innerHTML='';
  });

  return imgs;
}

function renderBadges(items){
  document.getElementById('summaryBadges').innerHTML=
    items.map(function(i){return '<span class="badge">'+escapeHtml(i)+'</span>';}).join('');
}
function renderTable(rows,title){
  var out=document.getElementById('output');
  if(!rows||!rows.length){out.innerHTML='<div class="section-title">'+escapeHtml(title)+'</div><p>Sin datos.</p>';return;}
  out.innerHTML='<div class="section-title">'+escapeHtml(title)+'</div>'+buildHtmlTable(rows);

function renderAnalysis(r){
  var s=r.stats||{};
  var h='<div class="section-title">Resumen estadístico</div><div class="stats-grid">';
  Object.keys(s).forEach(function(f){
    var st=s[f];
    h+='<div class="stat-card"><h4>'+escapeHtml(f)+'</h4>'+
      '<p><b>N:</b> '+escapeHtml(st.count)+'</p>'+
      '<p><b>Mín:</b> '+escapeHtml(st.min)+'  <b>Máx:</b> '+escapeHtml(st.max)+'</p>'+
      '<p><b>Media:</b> '+escapeHtml(st.mean)+'  <b>Med.:</b> '+escapeHtml(st.median)+'</p>'+
      '<p><b>Q1:</b> '+escapeHtml(st.q1)+'  <b>Q3:</b> '+escapeHtml(st.q3)+'</p>'+
      '<p><b>Desv.:</b> '+escapeHtml(st.stdDev)+'</p></div>';
  });
  h+='</div><div class="section-title">Gráficos</div><div class="charts-grid">'+
    ['chart_hist_n1:1. Histograma Nota 1','chart_hist_n2:2. Histograma Nota 2',
     'chart_hist_nf:3. Histograma Nota Final','chart_bar_grupos:4. Por Grupo',
     'chart_scatter_n1_nf:5. Nota 1 vs Nota Final','chart_scatter_n2_n3:6. Nota 2 vs Nota 3'
    ].map(function(s){var p=s.split(':');
      return '<div class="chart-card"><div class="chart-title">'+p[1]+'</div><div id="'+p[0]+'" class="chart-box"></div></div>';
    }).join('')+'</div>';
  if(r.preview&&r.preview.length)h+='<div class="section-title">Vista previa</div>'+buildHtmlTable(r.preview);
  document.getElementById('output').innerHTML=h;
  renderBadges(['Análisis OK','Registros: '+r.totalRows,'Gráficos: 6']);
  google.charts.setOnLoadCallback(function(){drawAllCharts(r.chartData||{});});
}

/* ══ Google Charts ══ */
function _opts(t,c){return{title:t,legend:{position:'none'},backgroundColor:'transparent',colors:[c],
  pointSize:6,titleTextStyle:{color:'#e5e7eb',fontSize:12},
  hAxis:{textStyle:{color:'#cbd5e1'},titleTextStyle:{color:'#cbd5e1'}},
  vAxis:{textStyle:{color:'#cbd5e1'},titleTextStyle:{color:'#cbd5e1'}},
  chartArea:{left:55,top:40,width:'80%',height:'68%'}};}
function drawAllCharts(cd){
  drawH('chart_hist_n1',   cd.histNota1,    'Histograma Nota 1',   '#60a5fa');
  drawH('chart_hist_n2',   cd.histNota2,    'Histograma Nota 2',   '#34d399');
  drawH('chart_hist_nf',   cd.histNotaFinal,'Histograma Nota Final','#f59e0b');
  drawB('chart_bar_grupos',cd.barGrupos,    'Por Grupo',           '#a78bfa');
  drawS('chart_scatter_n1_nf',cd.scatterN1Nf,'Nota 1 vs NF',      '#f87171');
  drawS('chart_scatter_n2_n3',cd.scatterN2N3,'Nota 2 vs Nota 3',  '#22d3ee');
}
function drawKMeansCharts(cd){
  drawB('chart_kmeans_bar',    cd.barClusters,    'Por cluster',   '#38bdf8');
  drawS('chart_kmeans_scatter',cd.scatterClusters,'NF por cluster','#facc15');
}
function drawH(id,rows,t,c){
  if(!rows||rows.length<=1)return;
  new google.visualization.Histogram(document.getElementById(id))
    .draw(google.visualization.arrayToDataTable(rows),Object.assign(_opts(t,c),{histogram:{bucketSize:0.5}}));
}
function drawB(id,rows,t,c){
  if(!rows||rows.length<=1)return;
  new google.visualization.ColumnChart(document.getElementById(id))
    .draw(google.visualization.arrayToDataTable(rows),_opts(t,c));
}
function drawS(id,rows,t,c){
  if(!rows||rows.length<=1)return;
  new google.visualization.ScatterChart(document.getElementById(id))
    .draw(google.visualization.arrayToDataTable(rows),_opts(t,c));
}

function buildHtmlTable(rows){
  if(!rows||!rows.length)return '<p>Sin datos.</p>';
  var hds=Object.keys(rows[0]);
  return '<div class="table-wrap"><table><thead><tr>'+
    hds.map(function(h){return '<th>'+escapeHtml(h)+'</th>';}).join('')+
    '</tr></thead><tbody>'+
    rows.map(function(row){return '<tr>'+hds.map(function(h){return '<td>'+escapeHtml(row[h])+'</td>';}).join('')+'</tr>';}).join('')+
    '</tbody></table></div>';
}
function escapeHtml(v){
  return String(v==null?'':v).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')
    .replace(/"/g,'&quot;').replace(/'/g,'&#039;');
}
